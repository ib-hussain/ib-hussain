We can fix the remaining issues. The main blocker is confirmed by your screenshot: you are **still in Wayland**, even after selecting GNOME. That explains why bare `Super` redirection is unreliable and why `xrandr` brightness still warns. We will now force a visible **GNOME Xorg (forced)** session, replace the Arch icon using your GitHub icon, remove the fake separator, apply macOS-like dock glass styling, bind `Super + A` to Quick Settings, and add a proper verifier. Dash-to-Panel is still the correct base because it combines launchers, running apps, and system tray indicators into one panel. ([GNOME Extensions][1])

Arch’s GDM behaviour can be affected by runtime config under `/run/gdm/custom.conf`, so this version removes that runtime override and creates a dedicated Xorg session entry instead of relying only on `WaylandEnable=false`. ([ArchWiki][2]) Dash-to-Panel’s own schema supports the settings we are using for bottom position, panel opacity, margins, radius, hover animation, and intellihide. ([GitHub][3])

WaylandEnable=false

from pathlib import Path

path = Path("/var/lib/AccountsService/users/ibrahim")
text = path.read_text() if path.exists() else ""

if "[User]" not in text:
    text = "[User]\n" + text

lines = []
for line in text.splitlines():
    if line.startswith("Session="):
        continue
    if line.startswith("XSession="):
        continue
    lines.append(line)

if "[User]" not in lines:
    lines.insert(0, "[User]")

lines.append("Session=gnome")
lines.append("XSession=ib-gnome-xorg")

path.write_text("\n".join(lines) + "\n")
PY

echo "Ensuring GDM starts normally."

sudo systemctl enable gdm.service
sudo systemctl set-default graphical.target

echo "File 1 complete."

        down) brightnessctl -d "$BACKLIGHT" set 10%- ;;

CURRENT="$(cat "$CURRENT_FILE" 2>/dev/null || echo 1.00)"

Paste:

```bash
#!/bin/bash
set -euo pipefail

OUT="$HOME/rice-v8-status.txt"

pass() {
    echo "PASS: $1" | tee -a "$OUT"
}

warn() {
    echo "WARN: $1" | tee -a "$OUT"
}

info() {
    echo "INFO: $1" | tee -a "$OUT"
}

check_command() {
    if command -v "$1" >/dev/null 2>&1; then
        pass "command exists: $1 -> $(command -v "$1")"
    else
        warn "missing command: $1"
    fi
}

check_pkg() {
    if pacman -Q "$1" >/dev/null 2>&1; then
        pass "package installed: $1"
    else
        warn "package missing: $1"
    fi
}

check_file() {
    if [ -e "$1" ]; then
        pass "file exists: $1"
    else
        warn "file missing: $1"
    fi
}

rm -f "$OUT"

echo "IB GNOME RICE V8 STATUS REPORT" | tee "$OUT"
echo "Generated: $(date)" | tee -a "$OUT"
echo "----------------------------------------" | tee -a "$OUT"

info "session type: ${XDG_SESSION_TYPE:-unknown}"

if [ "${XDG_SESSION_TYPE:-}" = "x11" ]; then
    pass "running under Xorg/X11"
else
    warn "still running under Wayland; reboot and select 'GNOME Xorg (forced)' from the gear menu"
fi

echo | tee -a "$OUT"
echo "GDM/session checks" | tee -a "$OUT"
check_file /usr/share/xsessions/ib-gnome-xorg.desktop
check_file /etc/gdm/custom.conf

if grep -q '^WaylandEnable=false' /etc/gdm/custom.conf 2>/dev/null; then
    pass "GDM WaylandEnable=false is set"
else
    warn "GDM WaylandEnable=false is not set"
fi

if grep -q '^DefaultSession=ib-gnome-xorg.desktop' /etc/gdm/custom.conf 2>/dev/null; then
    pass "GDM default session is ib-gnome-xorg.desktop"
else
    warn "GDM default session is not ib-gnome-xorg.desktop"
fi

if [ -f /run/gdm/custom.conf ]; then
    warn "/run/gdm/custom.conf exists and may override /etc/gdm/custom.conf"
else
    pass "no runtime /run/gdm/custom.conf override"
fi

echo | tee -a "$OUT"
echo "Package checks" | tee -a "$OUT"
for pkg in xorg-server gnome-shell gnome-session gnome-shell-extension-dash-to-panel gnome-shell-extension-appindicator gnome-shell-extension-desktop-icons-ng power-profiles-daemon rofi wofi xcape gnome-screenshot papirus-icon-theme; do
    check_pkg "$pkg"
done

echo | tee -a "$OUT"
echo "Command checks" | tee -a "$OUT"
for cmd in ib-app-search ib-power-mode ib-brightness gnome-screenshot rofi wofi xcape powerprofilesctl; do
    check_command "$cmd"
done

echo | tee -a "$OUT"
echo "GNOME extensions" | tee -a "$OUT"
gnome-extensions list --enabled | tee -a "$OUT" || true

if gnome-extensions list --enabled | grep -qx 'dash-to-panel@jderose9.github.com'; then
    pass "Dash-to-Panel enabled"
else
    warn "Dash-to-Panel not enabled"
fi

if gnome-extensions list --enabled | grep -qx 'dash-to-dock@micxgx.gmail.com'; then
    warn "Dash-to-Dock still enabled; it should be disabled"
else
    pass "Dash-to-Dock not enabled"
fi

if gnome-extensions list --enabled | grep -qx 'arcmenu@arcmenu.com'; then
    warn "ArcMenu still enabled; not needed because Arch icon is now a favourite launcher"
else
    pass "ArcMenu not enabled"
fi

echo | tee -a "$OUT"
echo "Dock settings" | tee -a "$OUT"

DTP="org.gnome.shell.extensions.dash-to-panel"

getsetting() {
    gsettings get "$1" "$2" 2>/dev/null || echo "unavailable"
}

info "panel-position: $(getsetting "$DTP" panel-position)"
info "panel-lengths: $(getsetting "$DTP" panel-lengths)"
info "panel-anchors: $(getsetting "$DTP" panel-anchors)"
info "panel-size: $(getsetting "$DTP" panel-size)"
info "panel-element-positions: $(getsetting "$DTP" panel-element-positions)"
info "trans-bg-color: $(getsetting "$DTP" trans-bg-color)"
info "trans-panel-opacity: $(getsetting "$DTP" trans-panel-opacity)"
info "global-border-radius: $(getsetting "$DTP" global-border-radius)"
info "intellihide: $(getsetting "$DTP" intellihide)"
info "intellihide-close-delay: $(getsetting "$DTP" intellihide-close-delay)"
info "animate-appicon-hover: $(getsetting "$DTP" animate-appicon-hover)"

FAVS="$(gsettings get org.gnome.shell favorite-apps 2>/dev/null || echo unavailable)"
info "favourite apps: $FAVS"

if echo "$FAVS" | grep -q "ib-arch-menu.desktop"; then
    pass "Arch menu launcher is in favourites"
else
    warn "Arch menu launcher is not in favourites"
fi

if echo "$FAVS" | grep -q "ib-dock-separator.desktop"; then
    warn "old fake separator still appears in favourites"
else
    pass "fake separator removed from favourites"
fi

if echo "$FAVS" | grep -q "ib-arch-menu.desktop.*ib-dock-separator.desktop"; then
    warn "separator still after Arch icon"
fi

echo | tee -a "$OUT"
echo "Arch icon checks" | tee -a "$OUT"
check_file "$HOME/.local/share/applications/ib-arch-menu.desktop"
if grep -q '^Icon=' "$HOME/.local/share/applications/ib-arch-menu.desktop" 2>/dev/null; then
    ICON_PATH="$(grep '^Icon=' "$HOME/.local/share/applications/ib-arch-menu.desktop" | cut -d= -f2-)"
    info "Arch icon path: $ICON_PATH"
    check_file "$ICON_PATH"
else
    warn "Arch launcher has no Icon field"
fi

echo | tee -a "$OUT"
echo "Keybinding checks" | tee -a "$OUT"
info "overlay key: $(getsetting org.gnome.mutter overlay-key)"
info "Super+D: $(getsetting org.gnome.desktop.wm.keybindings show-desktop)"
info "Screenshot UI: $(getsetting org.gnome.shell.keybindings show-screenshot-ui)"
info "Quick settings: $(getsetting org.gnome.shell.keybindings toggle-quick-settings)"
info "Custom keybindings: $(getsetting org.gnome.settings-daemon.plugins.media-keys custom-keybindings)"

if [ "${XDG_SESSION_TYPE:-}" = "x11" ] && pgrep -x xcape >/dev/null 2>&1; then
    pass "xcape is running for bare Super -> app search"
elif [ "${XDG_SESSION_TYPE:-}" = "x11" ]; then
    warn "Xorg is active but xcape is not running"
else
    warn "bare Super cannot be fully redirected until Xorg session is active"
fi

echo | tee -a "$OUT"
echo "Power profile checks" | tee -a "$OUT"
if systemctl is-active --quiet power-profiles-daemon.service; then
    pass "power-profiles-daemon is active"
else
    warn "power-profiles-daemon is not active"
fi

if command -v powerprofilesctl >/dev/null 2>&1; then
    info "current power profile: $(powerprofilesctl get 2>/dev/null || echo unavailable)"
    echo "Available profiles:" | tee -a "$OUT"
    powerprofilesctl list 2>/dev/null | tee -a "$OUT" || true
else
    warn "powerprofilesctl missing"
fi

echo | tee -a "$OUT"
echo "Desktop icon checks" | tee -a "$OUT"
info "DING show-home: $(getsetting org.gnome.shell.extensions.ding show-home)"
info "DING show-trash: $(getsetting org.gnome.shell.extensions.ding show-trash)"
info "DING show-volumes: $(getsetting org.gnome.shell.extensions.ding show-volumes)"

echo | tee -a "$OUT"
echo "Implemented / remaining summary" | tee -a "$OUT"

if [ "${XDG_SESSION_TYPE:-}" = "x11" ]; then
    pass "Implemented: Xorg active"
else
    warn "Remaining: choose GNOME Xorg (forced) at login"
fi

if echo "$FAVS" | grep -q "ib-arch-menu.desktop" && ! echo "$FAVS" | grep -q "ib-dock-separator.desktop"; then
    pass "Implemented: Arch launcher present and fake separator removed"
else
    warn "Remaining: dock favourite order needs correction"
fi

if [ "$(getsetting "$DTP" animate-appicon-hover)" = "false" ]; then
    pass "Implemented: icons should not jump outside dock"
else
    warn "Remaining: icon hover magnification still enabled"
fi

if [ "$(getsetting "$DTP" intellihide-close-delay)" = "10000" ]; then
    pass "Implemented: dock hide delay set to 10 seconds"
else
    warn "Remaining: dock hide delay is not 10000"
fi

if [ "$(getsetting org.gnome.shell.keybindings toggle-quick-settings)" != "unavailable" ]; then
    pass "Implemented: Super+A quick settings binding exists"
else
    warn "Remaining: this GNOME build does not expose toggle-quick-settings key"
fi

echo | tee -a "$OUT"
echo "Report saved to: $OUT" | tee -a "$OUT"
```
