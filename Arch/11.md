```bash
import St from 'gi://St';
import GLib from 'gi://GLib';
import Meta from 'gi://Meta';
import Clutter from 'gi://Clutter';
import {Extension} from 'resource:///org/gnome/shell/extensions/extension.js';
import * as Main from 'resource:///org/gnome/shell/ui/main.js';

export default class IbDesktopClock extends Extension {
    enable() {
        this._signals = [];
        this._timeout = 0;

        this._box = new St.BoxLayout({
            vertical: true,
            style_class: 'ib-desktop-clock-box',
            reactive: false,
            x_align: Clutter.ActorAlign.CENTER,
            y_align: Clutter.ActorAlign.CENTER
        });

        this._timeLabel = new St.Label({style_class: 'ib-desktop-clock-time'});
        this._dateLabel = new St.Label({style_class: 'ib-desktop-clock-date'});

        this._box.add_child(this._timeLabel);
        this._box.add_child(this._dateLabel);

        Main.layoutManager.uiGroup.add_child(this._box);

        this._signals.push(global.display.connect('window-created', () => this._queueSync()));
        this._signals.push(global.display.connect('notify::focus-window', () => this._queueSync()));
        this._signals.push(global.workspace_manager.connect('active-workspace-changed', () => this._queueSync()));

        this._timeout = GLib.timeout_add_seconds(GLib.PRIORITY_DEFAULT, 1, () => {
            this._sync();
            return GLib.SOURCE_CONTINUE;
        });

        this._sync();
    }

        if (this._timeout) {
            GLib.source_remove(this._timeout);
            this._timeout = 0;
        }

        if (this._signals) {

                try {
                    global.display.disconnect(id);
                } catch (_) {
                    try {
                        global.workspace_manager.disconnect(id);
                    } catch (_) {
                        try {
                            Main.layoutManager.disconnect(id);
                        } catch (_) {}
                    }
                }
            }
        }

        this._signals = [];

        if (this._box) {
            this._box.destroy();
            this._box = null;
        }
    }

    _queueSync() {
        GLib.timeout_add(GLib.PRIORITY_DEFAULT, 150, () => {
            this._sync();
            return GLib.SOURCE_REMOVE;
        });
    }

    _hasVisibleNormalWindow() {
        const workspace = global.workspace_manager.get_active_workspace();
        const windows = global.display.get_tab_list(Meta.TabList.NORMAL_ALL, workspace);

        for (const win of windows) {
            if (!win || win.minimized || win.skip_taskbar) {
                continue;
            }

            if (win.get_window_type && win.get_window_type() !== Meta.WindowType.NORMAL) {
                continue;
            }

            return true;
        }

        return false;
    }

    _sync() {
        if (!this._box) {
            return;
        }

        const now = GLib.DateTime.new_now_local();
        const timeText = now.format('%I:%M %p').replace(/^0/, '');
        const dateText = now.format('%A, %d %B %Y');

        this._timeLabel.set_text(timeText);
        this._dateLabel.set_text(dateText);

        const monitor = Main.layoutManager.primaryMonitor;
        this._box.set_position(
            Math.round(monitor.x + monitor.width / 2 - this._box.width / 2),
            Math.round(monitor.y + monitor.height * 0.18)
        );

        this._box.visible = !this._hasVisibleNormalWindow();
    }
}
EOF
```

```bash
import GLib from 'gi://GLib';

const TARGET_OPACITY = $TARGET_OPACITY;
const RESET_OPACITY = 255;

export default class IbWindowOpacity extends Extension {

        this._timeout = 0;

            return GLib.SOURCE_CONTINUE;

            this._timeout = 0;

                try {
                    global.display.disconnect(id);
                } catch (_) {}
            }
        }

        this._signals = [];
        this._reset();
    }

    _queueApply() {
        GLib.timeout_add(GLib.PRIORITY_DEFAULT, 250, () => {
            this._apply();
            return GLib.SOURCE_REMOVE;
        });

        if (!actor) {
            return false;

            return false;

            return false;

        return true;

    _apply() {
        for (const actor of global.get_window_actors()) {
            if (this._normalWindowActor(actor)) {
                actor.opacity = TARGET_OPACITY;
            }
        }
    }

    _reset() {
        for (const actor of global.get_window_actors()) {
            if (actor) {
                actor.opacity = RESET_OPACITY;
            }
        }
    }
}

#!/bin/bash
set -euo pipefail

set_governor() {
    local gov="$1"
    if [ -d /sys/devices/system/cpu/cpu0/cpufreq ]; then
        for f in /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor; do
            [ -e "$f" ] && echo "$gov" | sudo tee "$f" >/dev/null || true
        done
        return 0
    fi
    return 1
}

set_brightness_percent() {
    local pct="$1"
    if command -v brightnessctl >/dev/null 2>&1 && find /sys/class/backlight -mindepth 1 -maxdepth 1 | grep -q .; then
        brightnessctl set "${pct}%"
    fi
}

apply_profile() {
    local mode="$1"

    if command -v powerprofilesctl >/dev/null 2>&1 && powerprofilesctl list 2>/dev/null | grep -qE 'performance|balanced|power-saver'; then

            balanced) set_governor schedutil || set_governor ondemand || true ;;
            powersave|ultra) set_governor powersave || true ;;
        esac
    fi

    case "$mode" in
        ultra) set_brightness_percent 25 || true ;;
        powersave) set_brightness_percent 65 || true ;;
    esac

    notify-send "Power Mode" "Set to ${mode}"
}

else
    choice="$(printf "$menu" | rofi -dmenu -p "Power Mode")"
fi

case "$choice" in
    full|performance|balanced|powersave|ultra) apply_profile "$choice" ;;
esac
EOF
```

```bash
    info "terminal transparent background: $(gsettings get "$PROFILE_SCHEMA" use-transparent-background 2>/dev/null || echo unavailable)"
    info "terminal transparency percent: $(gsettings get "$PROFILE_SCHEMA" background-transparency-percent 2>/dev/null || echo unavailable)"
fi

echo | tee -a "$OUT"
echo "Desktop clock" | tee -a "$OUT"
if [ -d "$HOME/.local/share/gnome-shell/extensions/ib-desktop-clock@ibLaptop" ]; then
    pass "desktop clock extension installed"
else
    warn "desktop clock extension missing"
fi

echo | tee -a "$OUT"
echo "Power" | tee -a "$OUT"
if command -v powerprofilesctl >/dev/null 2>&1; then
    info "powerprofilesctl get: $(powerprofilesctl get 2>/dev/null || echo unavailable)"
    echo "Available profiles:" | tee -a "$OUT"
    powerprofilesctl list 2>/dev/null | tee -a "$OUT" || true
else
    warn "powerprofilesctl missing"
fi

if [ -d /sys/devices/system/cpu/cpu0/cpufreq ]; then
    pass "CPU governor interface exists"
    info "current governor: $(cat /sys/devices/system/cpu/cpu0/cpufreq/scaling_governor 2>/dev/null || echo unavailable)"
else
    warn "CPU governor interface not exposed; normal inside some VMs"
fi

echo | tee -a "$OUT"
echo "Report saved to: $OUT" | tee -a "$OUT"
```
