```text
Terminology  → GNOME Terminal
Nemo         → Nautilus / GNOME Files
Super+Enter  → removed
Ctrl+Alt+T   → opens GNOME Terminal
```

Nautilus is the file manager you are talking about: Ubuntu’s “Files” app is GNOME Files/Nautilus. Arch packages it as `nautilus`, described as the default GNOME file manager. GNOME Terminal is also available directly as `gnome-terminal`. ([Arch Linux][1]) ([Arch Linux][2])

For fonts, install Ubuntu’s own font family using Arch’s `ttf-ubuntu-font-family` package. ([Arch Linux][3])

For Bluetooth, install `bluez`, `bluez-utils`, and `blueman`, then enable `bluetooth.service`. ArchWiki’s Bluetooth page also notes that `rfkill` blocking should be checked before using Bluetooth. ([Arch Wiki][4])

Eww is still the correct dock layer because it is independent of the window manager and is made for custom widgets across window managers. ([elkowar.github.io][5])

# 5. Performance modes

I am going to implement your five modes as:

```text
full        = maximum allowed hardware state
performance = high performance with thermal throttling allowed
balanced    = normal
powersave   = reduced power, brightness stays full
ultra       = aggressive power saving + lower brightness
```

One technical truth: software cannot make RAM “fully used” while the machine is idle unless you run a workload that allocates memory. What this script does instead is remove artificial limits, reduce swapping in performance modes, and allow the workload to use RAM freely.

Create the controller:

```bash
sudo nano /usr/local/bin/ib-performance-mode
```

Paste:

```bash
#!/bin/bash

set -e

MODE="${1:-balanced}"
MODE_FILE="/var/lib/ib-performance-mode"

mkdir -p /var/lib

set_governor() {
    GOV="$1"
    for CPU_GOV in /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor; do
        [ -w "$CPU_GOV" ] && echo "$GOV" > "$CPU_GOV" || true
    done
}

set_epp() {
    EPP="$1"
    for CPU_EPP in /sys/devices/system/cpu/cpu*/cpufreq/energy_performance_preference; do
        [ -w "$CPU_EPP" ] && echo "$EPP" > "$CPU_EPP" || true
    done
}

set_min_max_freq() {
    MIN_POLICY="$1"

    for CPU_DIR in /sys/devices/system/cpu/cpu*/cpufreq; do
        [ -d "$CPU_DIR" ] || continue

        MIN_FILE="$CPU_DIR/scaling_min_freq"
        MAX_FILE="$CPU_DIR/scaling_max_freq"
        CPUINFO_MIN="$CPU_DIR/cpuinfo_min_freq"
        CPUINFO_MAX="$CPU_DIR/cpuinfo_max_freq"

        [ -r "$CPUINFO_MIN" ] || continue
        [ -r "$CPUINFO_MAX" ] || continue

        MIN_VAL=$(cat "$CPUINFO_MIN")
        MAX_VAL=$(cat "$CPUINFO_MAX")

        if [ "$MIN_POLICY" = "max" ]; then
            [ -w "$MIN_FILE" ] && echo "$MAX_VAL" > "$MIN_FILE" || true
            [ -w "$MAX_FILE" ] && echo "$MAX_VAL" > "$MAX_FILE" || true
        elif [ "$MIN_POLICY" = "normal" ]; then
            [ -w "$MIN_FILE" ] && echo "$MIN_VAL" > "$MIN_FILE" || true
            [ -w "$MAX_FILE" ] && echo "$MAX_VAL" > "$MAX_FILE" || true
        elif [ "$MIN_POLICY" = "low" ]; then
            [ -w "$MIN_FILE" ] && echo "$MIN_VAL" > "$MIN_FILE" || true
            [ -w "$MAX_FILE" ] && echo "$MAX_VAL" > "$MAX_FILE" || true
        fi
    done
}

set_brightness() {
    VALUE="$1"
    if command -v brightnessctl >/dev/null 2>&1; then
        brightnessctl set "$VALUE" >/dev/null 2>&1 || true
    fi
}

set_swappiness() {
    VALUE="$1"
    sysctl -w vm.swappiness="$VALUE" >/dev/null 2>&1 || true
}

set_dirty_ratios() {
    DIRTY_RATIO="$1"
    DIRTY_BACKGROUND="$2"
    sysctl -w vm.dirty_ratio="$DIRTY_RATIO" >/dev/null 2>&1 || true
    sysctl -w vm.dirty_background_ratio="$DIRTY_BACKGROUND" >/dev/null 2>&1 || true
}

set_pcie_policy() {
    POLICY="$1"
    if [ -w /sys/module/pcie_aspm/parameters/policy ]; then
        echo "$POLICY" > /sys/module/pcie_aspm/parameters/policy || true
    fi
}

set_fans_full() {
    for HWMON in /sys/class/hwmon/hwmon*; do
        for PWM_ENABLE in "$HWMON"/pwm*_enable; do
            [ -w "$PWM_ENABLE" ] && echo 1 > "$PWM_ENABLE" || true
        done

        for PWM in "$HWMON"/pwm[0-9]*; do
            [ -w "$PWM" ] && echo 255 > "$PWM" || true
        done
    done
}

set_fans_auto() {
    for HWMON in /sys/class/hwmon/hwmon*; do
        for PWM_ENABLE in "$HWMON"/pwm*_enable; do
            [ -w "$PWM_ENABLE" ] && echo 2 > "$PWM_ENABLE" || true
        done
    done
}

set_gpu_high() {
    for GPU_PERF in /sys/class/drm/card*/device/power_dpm_force_performance_level; do
        [ -w "$GPU_PERF" ] && echo high > "$GPU_PERF" || true
    done

    if command -v nvidia-smi >/dev/null 2>&1; then
        nvidia-smi -pm 1 >/dev/null 2>&1 || true
    fi

    if command -v nvidia-settings >/dev/null 2>&1; then
        nvidia-settings -a '[gpu:0]/GPUPowerMizerMode=1' >/dev/null 2>&1 || true
    fi
}

set_gpu_auto() {
    for GPU_PERF in /sys/class/drm/card*/device/power_dpm_force_performance_level; do
        [ -w "$GPU_PERF" ] && echo auto > "$GPU_PERF" || true
    done

    if command -v nvidia-settings >/dev/null 2>&1; then
        nvidia-settings -a '[gpu:0]/GPUPowerMizerMode=0' >/dev/null 2>&1 || true
    fi
}

set_gpu_low() {
    for GPU_PERF in /sys/class/drm/card*/device/power_dpm_force_performance_level; do
        [ -w "$GPU_PERF" ] && echo low > "$GPU_PERF" || true
    done
}

case "$MODE" in
    full)
        powerprofilesctl set performance >/dev/null 2>&1 || true
        set_governor performance
        set_epp performance
        set_min_max_freq max
        set_swappiness 1
        set_dirty_ratios 40 10
        set_pcie_policy performance
        set_brightness 100%
        set_fans_full
        set_gpu_high
        echo "full" > "$MODE_FILE"
        notify-send "Performance Mode" "Full mode: CPU/GPU set to maximum available state, fans forced if supported."
        ;;

    performance)
        powerprofilesctl set performance >/dev/null 2>&1 || true
        set_governor performance
        set_epp performance
        set_min_max_freq normal
        set_swappiness 10
        set_dirty_ratios 30 10
        set_pcie_policy performance
        set_brightness 100%
        set_fans_auto
        set_gpu_auto
        echo "performance" > "$MODE_FILE"
        notify-send "Performance Mode" "Performance mode: high performance with normal thermal control."
        ;;

    balanced)
        powerprofilesctl set balanced >/dev/null 2>&1 || true
        set_governor schedutil || set_governor ondemand || true
        set_epp balance_performance
        set_min_max_freq normal
        set_swappiness 60
        set_dirty_ratios 20 10
        set_pcie_policy default
        set_fans_auto
        set_gpu_auto
        echo "balanced" > "$MODE_FILE"
        notify-send "Performance Mode" "Balanced mode enabled."
        ;;

    powersave)
        powerprofilesctl set power-saver >/dev/null 2>&1 || true
        set_governor powersave
        set_epp balance_power
        set_min_max_freq low
        set_swappiness 80
        set_dirty_ratios 10 5
        set_pcie_policy powersave
        set_brightness 100%
        set_fans_auto
        set_gpu_low
        echo "powersave" > "$MODE_FILE"
        notify-send "Performance Mode" "Power-saving mode: reduced power, full brightness."
        ;;

    ultra)
        powerprofilesctl set power-saver >/dev/null 2>&1 || true
        set_governor powersave
        set_epp power
        set_min_max_freq low
        set_swappiness 100
        set_dirty_ratios 5 2
        set_pcie_policy powersupersave
        set_brightness 25%
        set_fans_auto
        set_gpu_low
        echo "ultra" > "$MODE_FILE"
        notify-send "Performance Mode" "Ultra power-saving mode enabled."
        ;;

    status)
        cat "$MODE_FILE" 2>/dev/null || echo "balanced"
        ;;

    *)
        echo "Usage: ib-performance-mode full|performance|balanced|powersave|ultra|status"
        exit 1
        ;;
esac
```

Make it executable:

```bash
sudo chmod +x /usr/local/bin/ib-performance-mode
```

Because your `ibrahim` user already has passwordless sudo, the dock can call this without asking for a password.

```bash
#!/bin/bash

button() {
    ICON="$1"
    NAME="$2"
    CMD="$3"
    CLASS_REGEX="$4"
    ACTIVE="$5"

    if [ "$ACTIVE" = "yes" ]; then
        printf '(button :class "dock-app active" :tooltip "%s" :onclick "~/.config/eww/scripts/app_action.sh '\''%s'\'' '\''%s'\'' '\''%s'\''" (box :orientation "v" :space-evenly false (label :class "app-icon" :text "%s") (label :class "app-dot" :text "●")))' "$NAME" "$NAME" "$CMD" "$CLASS_REGEX" "$ICON"
    else
        printf '(button :class "dock-app" :tooltip "%s" :onclick "~/.config/eww/scripts/app_action.sh '\''%s'\'' '\''%s'\'' '\''%s'\''" (box :orientation "v" :space-evenly false (label :class "app-icon" :text "%s") (label :class "app-dot-empty" :text "●")))' "$NAME" "$NAME" "$CMD" "$CLASS_REGEX" "$ICON"

}

is_running() {
    REGEX="$1"
    wmctrl -lx | grep -Eiq "$REGEX"
}

render() {
    WINDOWS=$(wmctrl -lx | awk '{print tolower($3)}' | sort -u)

    echo -n '(box :class "app-strip" :orientation "h" :space-evenly false :spacing 6 '

    if is_running "google-chrome|chromium"; then button "" "Chrome" "google-chrome-stable || chromium" "google-chrome|chromium" "yes"; else button "" "Chrome" "google-chrome-stable || chromium" "google-chrome|chromium" "no"; fi

    if is_running "gnome-terminal|gnome-terminal-server"; then button "" "Terminal" "gnome-terminal" "gnome-terminal|gnome-terminal-server" "yes"; else button "" "Terminal" "gnome-terminal" "gnome-terminal|gnome-terminal-server" "no"; fi

    if is_running "code|visual-studio-code"; then button "" "Code" "code" "code|visual-studio-code" "yes"; else button "" "Code" "code" "code|visual-studio-code" "no"; fi

    echo "$WINDOWS" | while read -r CLASS; do
        [ -z "$CLASS" ] && continue

        case "$CLASS" in
            *firefox*|*google-chrome*|*chromium*|*gnome-terminal*|*gnome-terminal-server*|*nautilus*|*org.gnome.nautilus*|*code*|*visual-studio-code*|*eww*|*rofi*)
                continue
                ;;
        esac

        CLEAN_NAME=$(echo "$CLASS" | awk -F'.' '{print $NF}' | sed 's/-/ /g' | sed 's/_/ /g')
        button "" "$CLEAN_NAME" "true" "$CLASS" "yes"
    done

    echo ')'
}

while true; do
    render
    sleep 2
done
```

```bash
IFACE=$(ip route | awk '/default/ {print $5; exit}')

if [ -z "$IFACE" ]; then

RX1=$(cat /sys/class/net/"$IFACE"/statistics/rx_bytes)

RX2=$(cat /sys/class/net/"$IFACE"/statistics/rx_bytes)
```

```bash
        ;;
    Suspend)

        ;;
```
