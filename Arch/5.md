#!/bin/bash

set -euo pipefail

DEV="${1:?device required}"
USER_NAME="ibrahim"
USER_HOME="/home/${USER_NAME}"
UID_NUM="$(id -u "$USER_NAME")"
GID_NUM="$(id -g "$USER_NAME")"

sleep 1

    exit 0
fi

FSTYPE="$(blkid -o value -s TYPE "$DEV" 2>/dev/null || true)"
LABEL="$(lsblk -no LABEL "$DEV" 2>/dev/null | head -n 1 | xargs || true)"
UUID="$(blkid -o value -s UUID "$DEV" 2>/dev/null || true)"

if [ -z "$FSTYPE" ]; then
    exit 0
fi

if [ -n "$LABEL" ]; then
    NAME="$LABEL"
elif [ -n "$UUID" ]; then
    NAME="$UUID"
else
    NAME="$(basename "$DEV")"
fi

SAFE_NAME="$(echo "$NAME" | tr -cd '[:alnum:] ._-' | sed 's/[[:space:]]\+/_/g')"

if [ -z "$SAFE_NAME" ]; then
    SAFE_NAME="$(basename "$DEV")"
fi

MNT="${USER_HOME}/${SAFE_NAME}"

if findmnt -rn "$DEV" >/dev/null 2>&1; then
    exit 0
fi

mkdir -p "$MNT"
chown "$USER_NAME:$USER_NAME" "$MNT"

case "$FSTYPE" in
    vfat|exfat|ntfs|ntfs3)
        mount -o "uid=${UID_NUM},gid=${GID_NUM},umask=022,rw" "$DEV" "$MNT"
        ;;
    *)
        mount "$DEV" "$MNT"
        chown "$USER_NAME:$USER_NAME" "$MNT" || true
        ;;
esac
EOF
