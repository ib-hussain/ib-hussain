#!/bin/bash

# ==========================================================
# ARCH LINUX INSTALLATION SCRIPT
# Desktop/WM: Enlightenment
# Display Manager: LightDM
# Default boot mode: UEFI/GPT
# Alternative: Legacy BIOS/MBR
# Python: 3.12.7 via pyenv
# Hostname: ibLaptop
# User: ibrahim with passwordless sudo
# ==========================================================

set -euo pipefail

# ==========================================================
# USER CONFIGURATION
# ==========================================================

DISK="/dev/sda"

INSTALL_MODE="uefi"
# Options:
#   uefi  = UEFI + GPT
#   bios  = Legacy BIOS + MBR

KERNEL_CHOICE="linux"
# Options:
#   linux       = current Arch kernel
#   linux-lts   = official Arch LTS kernel
#   lts515-aur  = install official linux-lts first, then optionally build linux-lts515 from AUR later

GPU_PROFILE="standard"
# Options:
#   standard = normal modern laptop/desktop
#   nouveau  = old NVIDIA laptop such as NVS 5200M

USER_NAME="ibrahim"
HOST_NAME="ibLaptop"
TIMEZONE="Asia/Karachi"
LOCALE="en_US.UTF-8"
PYTHON_VERSION="3.12.7"

# UEFI/GPT partitions
EFI_PART="/dev/sda1"
SWAP_PART="/dev/sda2"
ROOT_PART="/dev/sda3"

# BIOS/MBR partitions
BIOS_SWAP_PART="/dev/sda1"
BIOS_ROOT_PART="/dev/sda2"

# For your normal laptop:
# INSTALL_MODE="uefi"
# KERNEL_CHOICE="linux"
# GPU_PROFILE="standard"

# For Older Computer(like 2012 NVS 5200M laptop):
# INSTALL_MODE="bios"
# KERNEL_CHOICE="lts515-aur"
# GPU_PROFILE="nouveau"

# ==========================================================
# NETWORK IN LIVE ISO
# ==========================================================

echo "Connect Wi-Fi manually using iwctl if not already connected."
echo "Example:"
echo "  iwctl"
echo "  station wlan0 get-networks"
echo "  station wlan0 connect YOUR_WIFI_NAME"
echo "  exit"
echo

ping -c 4 archlinux.org

# ==========================================================
# SYNC LIVE ENVIRONMENT KEYRING
# ==========================================================

pacman -Sy --noconfirm archlinux-keyring

# ==========================================================
# PARTITIONING
# ==========================================================

if [[ "$INSTALL_MODE" == "uefi" ]]; then
    echo "UEFI/GPT mode selected."
    echo "Open cfdisk and create:"
    echo "  ${EFI_PART}  = 512M-1G EFI System"
    echo "  ${SWAP_PART} = 4G-8G Linux swap"
    echo "  ${ROOT_PART} = remaining Linux filesystem"
    cfdisk "$DISK"

    mkfs.fat -F32 "$EFI_PART"
    mkswap "$SWAP_PART"
    swapon "$SWAP_PART"
    mkfs.ext4 "$ROOT_PART"

    mount "$ROOT_PART" /mnt
    mkdir -p /mnt/boot
    mount "$EFI_PART" /mnt/boot

elif [[ "$INSTALL_MODE" == "bios" ]]; then
    echo "Legacy BIOS/MBR mode selected."
    echo "Open cfdisk and create a DOS/MBR partition table:"
    echo "  ${BIOS_SWAP_PART} = 4G-8G Linux swap"
    echo "  ${BIOS_ROOT_PART} = remaining Linux filesystem"
    echo "Mark ${BIOS_ROOT_PART} as bootable."
    cfdisk "$DISK"

    mkswap "$BIOS_SWAP_PART"
    swapon "$BIOS_SWAP_PART"
    mkfs.ext4 "$BIOS_ROOT_PART"

    mount "$BIOS_ROOT_PART" /mnt

else
    echo "Invalid INSTALL_MODE. Use 'uefi' or 'bios'."
    exit 1
fi

# ==========================================================
# BASE SYSTEM
# ==========================================================

if [[ "$KERNEL_CHOICE" == "linux" ]]; then
    pacstrap /mnt base linux linux-firmware
elif [[ "$KERNEL_CHOICE" == "linux-lts" ]]; then
    pacstrap /mnt base linux-lts linux-firmware
elif [[ "$KERNEL_CHOICE" == "lts515-aur" ]]; then
    echo "Using official linux-lts as a safe fallback first."
    echo "Exact 5.15 kernel must be built from AUR after base install."
    pacstrap /mnt base linux-lts linux-firmware
else
    echo "Invalid KERNEL_CHOICE."
    exit 1
fi

genfstab -U /mnt >> /mnt/etc/fstab

# ==========================================================
# CHROOT CONFIGURATION
# ==========================================================

arch-chroot /mnt /usr/bin/env \
    INSTALL_MODE="$INSTALL_MODE" \
    KERNEL_CHOICE="$KERNEL_CHOICE" \
    GPU_PROFILE="$GPU_PROFILE" \
    USER_NAME="$USER_NAME" \
    HOST_NAME="$HOST_NAME" \
    TIMEZONE="$TIMEZONE" \
    LOCALE="$LOCALE" \
    PYTHON_VERSION="$PYTHON_VERSION" \
    DISK="$DISK" \
    /bin/bash <<'CHROOT'

set -euo pipefail

# ==========================================================
# TIMEZONE AND CLOCK
# ==========================================================

ln -sf "/usr/share/zoneinfo/${TIMEZONE}" /etc/localtime
hwclock --systohc

# ==========================================================
# LOCALE GENERATION
# ==========================================================

sed -i "s/^#${LOCALE} UTF-8/${LOCALE} UTF-8/" /etc/locale.gen
locale-gen
echo "LANG=${LOCALE}" > /etc/locale.conf
echo "KEYMAP=us" > /etc/vconsole.conf

# ==========================================================
# HOSTNAME
# ==========================================================

echo "$HOST_NAME" > /etc/hostname

cat > /etc/hosts <<EOF
127.0.0.1   localhost
::1         localhost
127.0.1.1   ${HOST_NAME}.localdomain ${HOST_NAME}
EOF

# ==========================================================
# ROOT PASSWORD
# ==========================================================

echo "Set root password:"
passwd

# ==========================================================
# ESSENTIAL SYSTEM TOOLS
# Separate package groups to avoid random bulk installs
# ==========================================================

pacman -S --needed --noconfirm sudo
pacman -S --needed --noconfirm nano
pacman -S --needed --noconfirm git
pacman -S --needed --noconfirm base-devel

# Optional but useful for Intel laptops
pacman -S --needed --noconfirm intel-ucode

# ==========================================================
# USER SETUP
# ==========================================================

useradd -m -G wheel,video,audio,storage,optical,input "$USER_NAME"

echo "Set password for ${USER_NAME}:"
passwd "$USER_NAME"

# Full admin access through passwordless sudo.
# This is safer than making ibrahim UID 0.
cat > "/etc/sudoers.d/${USER_NAME}" <<EOF
${USER_NAME} ALL=(ALL:ALL) NOPASSWD: ALL
EOF

chmod 440 "/etc/sudoers.d/${USER_NAME}"

# Convenience aliases so common admin commands do not require typing sudo.
cat >> "/home/${USER_NAME}/.bashrc" <<'EOF'

# Admin shortcuts
alias root='sudo -i'
alias pacman='sudo pacman'
alias systemctl='sudo systemctl'
alias journalctl='sudo journalctl'
alias reboot='sudo reboot'
alias poweroff='sudo poweroff'
alias update='sudo pacman -Syu'
alias install='sudo pacman -S'
alias remove='sudo pacman -Rns'
EOF

chown "$USER_NAME:$USER_NAME" "/home/${USER_NAME}/.bashrc"

# ==========================================================
# BOOTLOADER
# ==========================================================

if [[ "$INSTALL_MODE" == "uefi" ]]; then
    pacman -S --needed --noconfirm grub
    pacman -S --needed --noconfirm efibootmgr
    pacman -S --needed --noconfirm os-prober

    sed -i 's/^#GRUB_DISABLE_OS_PROBER=false/GRUB_DISABLE_OS_PROBER=false/' /etc/default/grub || true
    grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=ArchLinux
    grub-mkconfig -o /boot/grub/grub.cfg

elif [[ "$INSTALL_MODE" == "bios" ]]; then
    pacman -S --needed --noconfirm grub
    pacman -S --needed --noconfirm os-prober

    sed -i 's/^#GRUB_DISABLE_OS_PROBER=false/GRUB_DISABLE_OS_PROBER=false/' /etc/default/grub || true
    grub-install --target=i386-pc "$DISK"
    grub-mkconfig -o /boot/grub/grub.cfg
fi

# ==========================================================
# NETWORK
# ==========================================================

pacman -S --needed --noconfirm networkmanager
systemctl enable NetworkManager

# ==========================================================
# GRAPHICS STACK
# ==========================================================

pacman -S --needed --noconfirm mesa

if [[ "$GPU_PROFILE" == "nouveau" ]]; then
    pacman -S --needed --noconfirm xf86-video-nouveau
fi

# ==========================================================
# XORG BASE
# Minimal Xorg stack for Enlightenment + LightDM
# ==========================================================

pacman -S --needed --noconfirm xorg-server
pacman -S --needed --noconfirm xorg-xinit
pacman -S --needed --noconfirm xorg-xrandr
pacman -S --needed --noconfirm xorg-xsetroot

# ==========================================================
# ENLIGHTENMENT DESKTOP / WINDOW MANAGER
# ==========================================================

pacman -S --needed --noconfirm enlightenment
pacman -S --needed --noconfirm terminology

# ==========================================================
# DISPLAY MANAGER
# ==========================================================

pacman -S --needed --noconfirm lightdm
pacman -S --needed --noconfirm lightdm-gtk-greeter

sed -i 's/^#greeter-session=.*/greeter-session=lightdm-gtk-greeter/' /etc/lightdm/lightdm.conf
systemctl enable lightdm

# ==========================================================
# PYENV + PYTHON 3.12.7
# Dependencies separated for clarity
# ==========================================================

pacman -S --needed --noconfirm pyenv

# Build dependencies needed by CPython through pyenv
pacman -S --needed --noconfirm openssl
pacman -S --needed --noconfirm zlib
pacman -S --needed --noconfirm xz
pacman -S --needed --noconfirm bzip2
pacman -S --needed --noconfirm libffi
pacman -S --needed --noconfirm readline
pacman -S --needed --noconfirm sqlite
pacman -S --needed --noconfirm tk
pacman -S --needed --noconfirm ncurses
pacman -S --needed --noconfirm curl

cat >> "/home/${USER_NAME}/.bashrc" <<'EOF'

# pyenv configuration
export PYENV_ROOT="$HOME/.pyenv"
export PATH="$PYENV_ROOT/bin:$PATH"

if command -v pyenv >/dev/null 2>&1; then
    eval "$(pyenv init - bash)"
fi
EOF

cat > "/home/${USER_NAME}/.bash_profile" <<'EOF'
[[ -f ~/.bashrc ]] && . ~/.bashrc
EOF

chown "$USER_NAME:$USER_NAME" "/home/${USER_NAME}/.bashrc" "/home/${USER_NAME}/.bash_profile"

su - "$USER_NAME" -c "PYENV_ROOT=\"\$HOME/.pyenv\" pyenv install -s ${PYTHON_VERSION}"
su - "$USER_NAME" -c "PYENV_ROOT=\"\$HOME/.pyenv\" pyenv global ${PYTHON_VERSION}"
su - "$USER_NAME" -c "PYENV_ROOT=\"\$HOME/.pyenv\" pyenv rehash"
su - "$USER_NAME" -c "PYENV_ROOT=\"\$HOME/.pyenv\" pyenv exec python --version"

# ==========================================================
# OPTIONAL: EXACT 5.15 LTS KERNEL FROM AUR
# For old NVIDIA proprietary 390xx experiments only.
# Nouveau does not require this by default.
# This is intentionally not enabled automatically unless KERNEL_CHOICE=lts515-aur.
# ==========================================================

if [[ "$KERNEL_CHOICE" == "lts515-aur" ]]; then
    echo "Building linux-lts515 from AUR. This may be heavy and is community-maintained."

    su - "$USER_NAME" -c "mkdir -p ~/aur"
    su - "$USER_NAME" -c "cd ~/aur && git clone https://aur.archlinux.org/linux-lts515.git || true"
    su - "$USER_NAME" -c "cd ~/aur/linux-lts515 && makepkg -s --noconfirm"

    mapfile -t LTS515_PKGS < <(find "/home/${USER_NAME}/aur/linux-lts515" -maxdepth 1 -name 'linux-lts515-*.pkg.tar.*' ! -name '*debug*' -print)

    if [[ "${#LTS515_PKGS[@]}" -gt 0 ]]; then
        pacman -U --noconfirm "${LTS515_PKGS[@]}"
        grub-mkconfig -o /boot/grub/grub.cfg
    else
        echo "No linux-lts515 package files found. Skipping installation."
    fi
fi

CHROOT

# ==========================================================
# FINISH
# ==========================================================

umount -R /mnt

echo "Installation complete."
echo "Remove installation media and reboot."

reboot
